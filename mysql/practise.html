select u.name as user_name,d.name as degree_name ,p.batch,u.phonenumber,u.emailid

from user u,degree d,profile p,role r where r.id=u.role_id and u.profile_id=p.id

and d.id=p.degree_id and r.name='Alumni'  order by u.name;
